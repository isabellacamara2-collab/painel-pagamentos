<script>

(function(){

  const el = id => document.getElementById(id);

  const fmtBRL = v => (new Intl.NumberFormat('pt-BR', {style:'currency', currency:'BRL'})).format(v || 0);

  const now = () => new Date().getTime();

  const LS_KEY = 'notion-like-payments-jobs-v1';

  let jobs = [];

  let sortKey = 'data-asc';

  // Estado de filtros

  const state = {

    q: '',

    filtraData: 'todos',

    filtraEntrada: 'todos',

    filtraRestante: 'todos',

    ordena: 'data-asc'

  };

  // Util

  function parseNumberBR(str){

    if (typeof str === 'number') return str;

    if (!str) return 0;

    const s = String(str).replace(/\./g,'').replace(',','.');

    const val = parseFloat(s);

    return isNaN(val) ? 0 : val;

  }

  function uuid(){

    return 'id-' + Math.random().toString(36).slice(2) + Date.now().toString(36);

  }

  function load(){

    try{

      const raw = localStorage.getItem(LS_KEY);

      jobs = raw ? JSON.parse(raw) : [];

    }catch(e){ jobs = []; }

  }

  function save(){

    localStorage.setItem(LS_KEY, JSON.stringify(jobs));

  }

  // Cálculos

  function compute(job){

    const entradaValor = job.valorTotal * 0.5;

    const restanteValor = job.valorTotal - entradaValor;

    const isPast = Date.now() > new Date(job.dataISO).getTime();

    const entradaPago = job.statusEntrada === 'Pago';

    const restantePago = job.statusRestante === 'Pago';

    // Em aberto (para visão geral): soma do que falta pagar agora

    const abertoEntrada = entradaPago ? 0 : entradaValor;

    const abertoRestante = restantePago ? 0 : restanteValor;

    const emAberto = abertoEntrada + abertoRestante;

    // Progresso: 0, 50, 100

    let progresso = 0;

    if (entradaPago) progresso += 50;

    if (restantePago) progresso += 50;

    // Estados "atrasado" calculados (sem sobrescrever selects)

    const entradaAtrasada = isPast && !entradaPago;

    const restanteAtrasado = isPast && !restantePago;

    return {

      entradaValor, restanteValor, emAberto, progresso, isPast,

      entradaAtrasada, restanteAtrasada

    };

  }

  // Renderização

  function render(){

    // Filtros

    const q = state.q.trim().toLowerCase();

    let list = jobs.filter(j => {

      const c = compute(j);

      // Texto

      const text = (j.tipo + ' ' + j.local).toLowerCase();

      if (q && !text.includes(q)) return false;

      // Data

      if (state.filtraData === 'futuros' && c.isPast) return false;

      if (state.filtraData === 'passados' && !c.isPast) return false;

      // Entrada

      if (state.filtraEntrada !== 'todos'){

        if (state.filtraEntrada === 'pago' && j.statusEntrada !== 'Pago') return false;

        if (state.filtraEntrada === 'pendente' && j.statusEntrada !== 'Pendente') return false;

        if (state.filtraEntrada === 'atrasado' && !c.entradaAtrasada) return false;

      }

      // Restante

      if (state.filtraRestante !== 'todos'){

        if (state.filtraRestante === 'pago' && j.statusRestante !== 'Pago') return false;

        if (state.filtraRestante === 'pendente' && j.statusRestante !== 'Pendente') return false;

        if (state.filtraRestante === 'atrasado' && !c.restanteAtrasada) return false;

      }

      return true;

    });

    // Ordenação

    list.sort((a,b) => {

      const ca = compute(a), cb = compute(b);

      switch(state.ordena){

        case 'data-asc': return new Date(a.dataISO) - new Date(b.dataISO);

        case 'data-desc': return new Date(b.dataISO) - new Date(a.dataISO);

        case 'valor-asc': return a.valorTotal - b.valorTotal;

        case 'valor-desc': return b.valorTotal - a.valorTotal;

        case 'progresso-asc': return ca.progresso - cb.progresso;

        case 'progresso-desc': return cb.progresso - ca.progresso;

        default: return 0;

      }

    });

    // Visão Geral

    const bruto = jobs.reduce((acc,j)=> acc + j.valorTotal, 0);

    const aberto = jobs.reduce((acc,j)=> acc + compute(j).emAberto, 0);

    const agendados = jobs.filter(j => !compute(j).isPast).length;

    el('statBruto').textContent = fmtBRL(bruto);

    el('statAberto').textContent = fmtBRL(aberto);

    el('statAgendados').textContent = String(agendados);

    el('statTodos').textContent = String(jobs.length);

    // Desktop table

    const body = el('jobsBody');

    body.innerHTML = '';

    list.forEach(j => {

      const c = compute(j);

      const row = document.createElement('div');

      row.className = 'row' + ((c.entradaAtrasada || c.restanteAtrasada) ? ' overdue' : '');

      const cel1 = document.createElement('div'); cel1.className='cell';

      cel1.innerHTML = `<div>

        <div style="font-weight:700">${escapeHtml(j.tipo || '-')}</div>

        <div class="tiny mono">ID: ${j.id}</div>

      </div>`;

      const cel2 = document.createElement('div'); cel2.className='cell';

      cel2.textContent = j.local || '-';

      const cel3 = document.createElement('div'); cel3.className='cell';

      cel3.innerHTML = `

        <div class="mono">${fmtDateTime(j.dataISO)}</div>

        <div class="tiny">Faltam: <span class="countdown" data-id="${j.id}">--</span></div>

      `;

      const cel4 = document.createElement('div'); cel4.className='cell mono';

      cel4.textContent = fmtBRL(j.valorTotal);

      const cel5 = document.createElement('div'); cel5.className='cell mono';

      cel5.textContent = fmtBRL(c.entradaValor);

      const cel6 = document.createElement('div'); cel6.className='cell';

      cel6.appendChild(buildStatusSelect(j.id, 'entrada', j.statusEntrada, c.entradaAtrasada));

      const cel7 = document.createElement('div'); cel7.className='cell mono';

      cel7.textContent = fmtBRL(c.restanteValor);

      const cel8 = document.createElement('div'); cel8.className='cell';

      cel
